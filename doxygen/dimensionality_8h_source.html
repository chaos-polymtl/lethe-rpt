11namespace Parameters {
12/**
13 * @brief Class used to rescale physical properties for cases where the main
14 * dimensions of the problems (Length, Mass, Time and Temperature) are not
15 * expressed in SI, but are expressed in another set of units. For example, this
16 * can be used to carry out a simulation where the mesh is expressed in
17 * millimeters instead of meters. This class is currently only used to rescale
18 * the physical properties in the FEM-based physics (e.g. all of those that are
19 * solvers). Support for DEM, CFD-DEM and rescaling of boundary conditions is
20 * planned in the midterm.
21 */
22class Dimensionality {
23public:
24 /**
25 * @brief
26 * Default constructor that assumes that the units used in the simulation
27 * are consistent.
28 */
29 Dimensionality() : length(1), mass(1), time(1), temperature(1) {
30 define_all_scales();
31 }
32
33 /**
34 * @brief
35 * Defines all the sub-scales that are relevant to simulations. This is used
36 * to define scaling laws for kinematic viscosity or other quantities which
37 * appear multiple times in the simulation. This prevents the recalculation
38 * of these scales (and thus, human errors).
39 */
40 void define_all_scales();
41
42 /**
43 * @brief
44 * Declares the fundamental dimensions of the parameter file
45 *
46 * @param prm The parameter handler being used
47 */
48 static void declare_parameters(ParameterHandler &prm);
49
50 /**
51 * @brief
52 * Parses the fundamental dimensions from the parameter file
53 *
54 * @param prm The parameter handler being used
55 */
56 void parse_parameters(ParameterHandler &prm);
57
58 /*
59 * Fundamental dimensions. These are the dimensions that are read from the
60 * parameter file
61 */
62 double length;
63 double mass;
64 double time;
65 double temperature;
66
67 /*
68 * Scaling laws. These are constants that are used to rescale quantities
69 * arising frequently. Some of these quantities are not trivial to calculate
70 * (e.g. thermal conductivity). Thus, it is better to pre-calculate these
71 * scalings and to reuse them instead of calculating them on the fly
72 * everywhere.
73 */
74 double density_scaling;
75 double specific_gas_constant_scaling;
76 double viscosity_scaling;
77 double specific_heat_scaling;
78 double thermal_conductivity_scaling;
79 double enthalpy_scaling;
80 double diffusivity_scaling;
81 double thermal_expansion_scaling;
82 double surface_tension_scaling;
83 double surface_tension_gradient_scaling;
84 double cahn_hilliard_mobility_scaling;
85 double cahn_hilliard_epsilon_scaling;
86};
87} // namespace Parameters
