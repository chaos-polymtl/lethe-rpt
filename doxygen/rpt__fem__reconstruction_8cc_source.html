39template <int dim> void RPTL2Projection<dim>::setup_triangulation() {
40 if (fem_reconstruction_parameters.mesh_type ==
41 Parameters::RPTFEMReconstructionParameters::FEMMeshType::gmsh) {
42 // Warn user
43 pcout << "-----------------------------------------------" << std::endl;
44 pcout << "At the moment, the application calculates" << std::endl;
45 pcout << "the gamma-ray attenuation in cylindrical" << std::endl;
46 pcout << "vessels only. Using another type of geometry" << std::endl;
47 pcout << "would yield in committing an error when" << std::endl;
48 pcout << "calculating photon counts. Furthermore, the" << std::endl;
49 pcout << "axis of symmetry of the cylinder should" << std::endl;
50 pcout << "be oriented in z-direction." << std::endl;
51 pcout << "-----------------------------------------------" << std::endl;
52
53 // Import and attach mesh to triangulation
54 GridIn<dim> grid_in;
55 grid_in.attach_triangulation(triangulation);
56 std::ifstream input_file(fem_reconstruction_parameters.mesh_file);
57 grid_in.read_msh(input_file);
58
59 // Attach a cylindrical manifold object to the triangulation
60 const CylindricalManifold<dim> manifold(2); // Cylinder along the z-axis
61 triangulation.set_all_manifold_ids(0);
62 triangulation.set_manifold(0, manifold);
63
64 // Check if all cells are tetrahedral elements
65 for (const auto &cell : triangulation.active_cell_iterators()) {
66 AssertThrow(
67 cell->n_vertices() == 4,
68 ExcMessage(
69 "The imported mesh has to have only tetrahedral elements"));
70 }
71 } else {
72 Triangulation<dim> temp_triangulation;
73 Triangulation<dim> flat_temp_triangulation;
74
75 GridGenerator::subdivided_cylinder(
76 temp_triangulation, fem_reconstruction_parameters.z_subdivisions,
77 parameters.reactor_radius, parameters.reactor_height * 0.5);
78 temp_triangulation.refine_global(
79 fem_reconstruction_parameters.mesh_refinement);
80
81 // Flatten the triangulation
82 GridGenerator::flatten_triangulation(temp_triangulation,
83 flat_temp_triangulation);
84 // Convert to simplex elements
85 GridGenerator::convert_hypercube_to_simplex_mesh(flat_temp_triangulation,
86 triangulation);
87 triangulation.set_all_manifold_ids(0);
88
89 // Grid transformation
90 Tensor<1, dim, double> axis({0, 1, 0});
91 GridTools::rotate(axis, M_PI_2, triangulation);
92 Tensor<1, dim> shift_vector({0, 0, parameters.reactor_height * 0.5});
93 GridTools::shift(shift_vector, triangulation);
94 }
95}

159template <int dim> void RPTL2Projection<dim>::setup_system() {
160 TimerOutput::Scope t(computing_timer, "setup_system");
161
162 // Partition triangulation between the processes
163 GridTools::partition_triangulation(
164 Utilities::MPI::n_mpi_processes(mpi_communicator), triangulation);
165
166 dof_handler.distribute_dofs(fe);
167 DoFRenumbering::subdomain_wise(dof_handler);
168
169 constraints.clear();
170 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
171 constraints.close();
172
173 DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
174 DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints, false);
175
176 const std::vector<IndexSet> locally_owned_dofs_per_proc =
177 DoFTools::locally_owned_dofs_per_subdomain(dof_handler);
178
179 // Get the DoFs owned by each process
180 const IndexSet &locally_owned_dofs =
181 locally_owned_dofs_per_proc[Utilities::MPI::this_mpi_process(
182 mpi_communicator)];
183
184 nodal_counts.resize(n_detector);
185 for (Vector<double> &nodal_counts_for_one_detector : nodal_counts)
186 nodal_counts_for_one_detector.reinit(locally_owned_dofs.size());
187
188 system_matrix.reinit(locally_owned_dofs, locally_owned_dofs, dsp,
189 mpi_communicator);
190 solution.reinit(locally_owned_dofs, mpi_communicator);
191 system_rhs.reinit(locally_owned_dofs, mpi_communicator);
192}

194template <int dim>
195void RPTL2Projection<dim>::solve_linear_system(unsigned detector_no) {
196 TimerOutput::Scope t(computing_timer, "solve_linear_system");
197
198 this->pcout << "Norm of RHS is : " << system_rhs.l2_norm() << std::endl;
199
200 SolverControl solver_control(10000000,
201 std::max(1e-6, system_rhs.l2_norm() * 1e-6));
202 LA::SolverGMRES solver(solver_control);
203
204 LA::MPI::PreconditionILU preconditioner;
205 LA::MPI::PreconditionILU::AdditionalData data(0, 1e-10, 1, 0);
206 preconditioner.initialize(system_matrix, data);
207
208 solver.solve(system_matrix, solution, system_rhs, preconditioner);
209
210 Vector<double> localized_solution(
211 solution); // constructor Trilinos MPI Vector -> deal.II Vector
212
213 constraints.distribute(localized_solution);
214 nodal_counts[detector_no] = localized_solution;
215}

217template <int dim>
218void RPTL2Projection<dim>::assemble_system(unsigned no_detector) {
219 TimerOutput::Scope t(computing_timer, "assemble_system");
220
221 system_rhs = 0;
222 system_matrix = 0;
223
224 const QGaussSimplex<dim> quadrature_formula(fe.degree + 1);
225
226 FEValues<dim> fe_values(mapping, fe, quadrature_formula,
227 update_values | update_quadrature_points |
228 update_JxW_values);
229
230 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
231
232 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
233 Vector<double> cell_rhs(dofs_per_cell);
234
235 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
236
237 for (const auto &cell : dof_handler.active_cell_iterators()) {
238 if (cell->subdomain_id() ==
239 Utilities::MPI::this_mpi_process(mpi_communicator)) {
240 cell_matrix = 0;
241 cell_rhs = 0;
242 fe_values.reinit(cell);
243
244 for (const unsigned int q_index : fe_values.quadrature_point_indices()) {
245 Point<dim> q_point_position = fe_values.quadrature_point(q_index);
246 RadioParticle<dim> particle(q_point_position, 0);
247
248 ParticleDetectorInteractions<dim> p_q_interaction(
249 particle, detectors[no_detector], parameters);
250
251 double count = p_q_interaction.calculate_count();
252
253 if (count < 0 || count != count)
254 this->pcout << "Invalid value encountered in count calculation, this "
255 "should be investigated furthermore : "
256 << count <<
257
258 std::endl;
259 else {
260 for (const unsigned int i : fe_values.dof_indices()) {
261 for (const unsigned int j : fe_values.dof_indices()) {
262 cell_matrix(i, j) +=
263 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
264 fe_values.shape_value(j, q_index) * // phi_j(x_q)
265 fe_values.JxW(q_index)); // dx
266 }
267
268 cell_rhs(i) += (count * // f(x)
269 fe_values.shape_value(i, q_index) * // phi_i(x_q)
270 fe_values.JxW(q_index)); // dx
271 }
272 }
273 }
274 cell->get_dof_indices(local_dof_indices);
275 constraints.distribute_local_to_global(
276 cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
277 }
278 }
279 system_matrix.compress(VectorOperation::add);
280 system_rhs.compress(VectorOperation::add);
281}

425template <int dim>
426Vector<double> assemble_matrix_and_rhs(
427 std::vector<std::vector<double>> &vertex_count,
428 std::vector<double> &experimental_count,
429 Parameters::RPTFEMReconstructionParameters::FEMCostFunction
430 &cost_function_type) {
431 Vector<double> reference_location;
432 unsigned int detector_size = vertex_count.size();
433 double sigma = 0;
434 LAPACKFullMatrix<double> sys_matrix;
435 Vector<double> sys_rhs;
436
437 sys_matrix.reinit(dim);
438 sys_rhs.reinit(dim);
439
440 if (cost_function_type ==
441 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute) {
442 // Assembling sys_matrix (Jacobian)
443 for (int i = 0; i < dim; ++i) {
444 for (int j = 0; j < dim; ++j) {
445 for (unsigned int d = 0; d < detector_size; ++d) {
446 sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
447 (-vertex_count[d][0] + vertex_count[d][i + 1]);
448 }
449 sys_matrix.set(i, j, sigma);
450 sigma = 0;
451 }
452 }
453
454 // Assembling sys_rhs
455 for (int i = 0; i < dim; ++i) {
456 for (unsigned int d = 0; d < detector_size; ++d) {
457 sigma += (vertex_count[d][0] - experimental_count[d]) *
458 (-vertex_count[d][0] + vertex_count[d][i + 1]);
459 }
460 sys_rhs[i] = -sigma;
461 sigma = 0;
462 }
463 } else if (cost_function_type == Parameters::RPTFEMReconstructionParameters::
464 FEMCostFunction::relative) {
465 std::vector<double> denom(detector_size);
466
467 for (unsigned int d = 0; d < experimental_count.size(); ++d)
468 denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
469
470 // Assembling sys_matrix
471 for (int i = 0; i < dim; ++i) {
472 for (int j = 0; j < dim; ++j) {
473 for (unsigned int d = 0; d < detector_size; ++d) {
474 sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
475 (-vertex_count[d][0] + vertex_count[d][i + 1]) * denom[d];
476 }
477 sys_matrix.set(i, j, sigma);
478 sigma = 0;
479 }
480 }
481
482 // Assembling sys_rhs
483 for (int i = 0; i < dim; ++i) {
484 for (unsigned int d = 0; d < detector_size; ++d) {
485 sigma += (vertex_count[d][0] - experimental_count[d]) *
486 (-vertex_count[d][0] + vertex_count[d][i + 1]) * denom[d];
487 }
488 sys_rhs[i] = -sigma;
489 sigma = 0;
490 }
491 }
492
493 // Setup and solve linear system
494 sys_matrix.set_property(LAPACKSupport::general);
495 sys_matrix.compute_lu_factorization();
496 sys_matrix.solve(sys_rhs);
497
498 reference_location = sys_rhs;
499 return reference_location;
500}

502template <int dim>
503double RPTFEMReconstruction<dim>::calculate_reference_location_error(
504 Vector<double> &reference_location, const double &last_constraint) {
505 std::vector<double> err_coordinates(dim + 1, 0);
506 double norm_error_coordinates = 0;
507
508 // Check if the location is a valid one
509 for (int i = 0; i < dim; ++i) {
510 // e_1 = \xi - 1; e_2 = \eta - 1; e_3 = \zeta - 1;
511 if (reference_location[i] > 1)
512 err_coordinates[i] = (reference_location[i] - 1);
513
514 // e_1 = - \xi; e_2 = - \eta; e_3 = - \zeta;
515 if (reference_location[i] < 0)
516 err_coordinates[i] = std::abs(reference_location[i]);
517 }
518
519 // Check the last constraint
520 if (last_constraint < 0) {
521 // e_4 = |(1 - \xi - \eta - \zeta)|
522 err_coordinates[3] = std::abs(last_constraint);
523 } else if (last_constraint > 1) {
524 // e_4 = (1 - \xi - \eta - \zeta) - 1
525 err_coordinates[3] = last_constraint - 1;
526 }
527
528 // Calculate the norm of the error coordinate vector
529 for (const double &error : err_coordinates) {
530 norm_error_coordinates += error * error;
531 }
532
533 return norm_error_coordinates;
534}

536template <int dim>
537double RPTFEMReconstruction<dim>::calculate_cost(
538 const TriaActiveIterator<DoFCellAccessor<dim, dim, false>> &cell,
539 Vector<double> &reference_location, const double &last_constraint,
540 std::vector<double> &experimental_count) {
541 double cost = 0;
542 double count = 0;
543
544 if (fem_reconstruction_parameters.fem_cost_function ==
545 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute) {
546 // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
547 // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2
548 for (unsigned int d = 0; d < n_detector; ++d) {
549 count +=
550 (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
551
552 for (int i = 0; i < dim; ++i) {
553 count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
554 reference_location[i]);
555 }
556 count -= experimental_count[d];
557 cost += count * count;
558 count = 0;
559 }
560 } else if (fem_reconstruction_parameters.fem_cost_function ==
561 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::
562 relative) {
563 std::vector<double> denom(n_detector);
564
565 // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
566 // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2 / (
567 // C_{exp,d}^2)
568 for (unsigned int d = 0; d < n_detector; ++d) {
569 count +=
570 (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
571
572 denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
573
574 for (int i = 0; i < dim; ++i) {
575 count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
576 reference_location[i]);
577 }
578
579 count -= experimental_count[d];
580 cost += count * count * denom[d];
581 count = 0;
582 }
583 }
584
585 return cost;
586}

588template <int dim>
589void RPTFEMReconstruction<dim>::search_position_in_reference_space(
590 std::vector<std::vector<double>> &count_from_all_detectors,
591 std::vector<double> &experimental_count,
592 const typename DoFHandler<dim>::active_cell_iterator &cell,
593 const double &tol_reference_location, double &max_cost,
594 Vector<double> &reference_location, bool &position_found,
595 Point<dim> &real_location) {
596 double last_constraint_reference_location;
597 double norm_error_coordinates;
598 double calculated_cost;
599
600 // Solve linear system to find the location in reference coordinates
601 reference_location = assemble_matrix_and_rhs<dim>(
602 count_from_all_detectors, experimental_count,
603 fem_reconstruction_parameters.fem_cost_function);
604
605 // 4th constraint on the location of the particle in reference coordinates
606 last_constraint_reference_location =
607 1 - reference_location[0] - reference_location[1] - reference_location[2];
608
609 // Evaluate the error of the reference position (Is it outside the
610 // reference tetrahedron ?)
611 norm_error_coordinates = calculate_reference_location_error(
612 reference_location, last_constraint_reference_location);
613
614 // Extrapolation limit
615 if (norm_error_coordinates < tol_reference_location) {
616 // Calculate cost with the selected cost function
617 calculated_cost =
618 calculate_cost(cell, reference_location,
619 last_constraint_reference_location, experimental_count);
620 // Check if the new cost is lower than the previously stored
621 // real_location's
622 if (calculated_cost < max_cost) {
623 max_cost = calculated_cost;
624 position_found = true;
625 previous_position_cell = cell;
626
627 // Evaluate the real location of the particle
628 for (unsigned int v = 0; v < cell->n_vertices(); ++v) {
629 if (v == 0)
630 real_location = cell->vertex(v);
631 else
632 real_location +=
633 reference_location[v - 1] * (cell->vertex(v) - cell->vertex(0));
634 }
635 }
636 }
637}

639template <int dim>
640bool RPTFEMReconstruction<dim>::find_position_global_search(
641 std::vector<double> &experimental_counts,
642 const double tol_reference_location) {
643 double max_cost = DBL_MAX;
644 bool position_found = false;
645 Point<dim> real_location;
646 Vector<double> reference_location;
647 std::vector<std::vector<double>> count_from_all_detectors(
648 n_detector, std::vector<double>(4));
649
650 // Loop over cell, loop over detectors and get nodal count values for each
651 // vertex of the cell
652 for (const auto &cell : dof_handler.active_cell_iterators()) {
653 for (unsigned int d = 0; d < n_detector; ++d) {
654 for (unsigned int v = 0; v < cell->n_vertices(); ++v) {
655 auto dof_index = cell->vertex_dof_index(v, 0);
656 count_from_all_detectors[d][v] = nodal_counts[d][dof_index];
657 }
658 }
659 search_position_in_reference_space(
660 count_from_all_detectors, experimental_counts, cell,
661 tol_reference_location, max_cost, reference_location, position_found,
662 real_location);
663 }
664 found_positions.push_back(real_location);
665 return position_found;
666}

668template <int dim>
669bool RPTFEMReconstruction<dim>::find_position_local_search(
670 std::vector<double> &experimental_counts,
671 const double tol_reference_location) {
672 double max_cost = DBL_MAX;
673 bool position_found = false;
674 Point<dim> real_location;
675 Vector<double> reference_location;
676 std::vector<std::vector<double>> count_from_all_detectors(
677 n_detector, std::vector<double>(4));
678 const typename DoFHandler<dim>::active_cell_iterator cell =
679 previous_position_cell;
680
681 // Find cells adjacent to the cell were found the previous position
682 std::vector<typename DoFHandler<dim>::active_cell_iterator> adjacent_cells;
683 std::set<typename DoFHandler<dim>::active_cell_iterator> all_adjacent_cells;
684
685 // Add previous solution's cell
686 all_adjacent_cells.insert(cell);
687
688 // Search for adjacent cells and stock them in the set container
689 for (unsigned int v = 0; v < cell->n_vertices(); ++v) {
690 auto v_index = cell->vertex_index(v);
691 adjacent_cells =
692 GridTools::find_cells_adjacent_to_vertex(dof_handler, v_index);
693
694 for (const auto &adjacent_cell : adjacent_cells)
695 all_adjacent_cells.insert(adjacent_cell);
696 }
697
698 // Add adjacent cells according to the level of proximity
699 if (fem_reconstruction_parameters.search_proximity_level > 1) {
700 std::set<typename DoFHandler<dim>::active_cell_iterator>
701 previous_adjacent_cells;
702 std::set<typename DoFHandler<dim>::active_cell_iterator>
703 previous_main_cells;
704
705 previous_main_cells.insert(cell);
706
707 for (unsigned int proximity_level = 2;
708 proximity_level <
709 fem_reconstruction_parameters.search_proximity_level + 1;
710 ++proximity_level) {
711 previous_adjacent_cells = all_adjacent_cells;
712
713 for (const auto &current_main_cell : previous_adjacent_cells) {
714 if (*previous_main_cells.find(current_main_cell) != current_main_cell) {
715 for (unsigned int v = 0; v < current_main_cell->n_vertices(); ++v) {
716 auto v_index = current_main_cell->vertex_index(v);
717 adjacent_cells =
718 GridTools::find_cells_adjacent_to_vertex(dof_handler, v_index);
719
720 for (const auto &adjacent_cell : adjacent_cells)
721 all_adjacent_cells.insert(adjacent_cell);
722 }
723 }
724 previous_main_cells = previous_adjacent_cells;
725 }
726 }
727 }
728
729 // Loop over adjacent cells, loop over detectors and get nodal count values
730 // for each vertex of the cell
731 for (const auto &adjacent_cell : all_adjacent_cells) {
732 for (unsigned int d = 0; d < n_detector; ++d) {
733 for (unsigned int v = 0; v < adjacent_cell->n_vertices(); ++v) {
734 auto dof_index = adjacent_cell->vertex_dof_index(v, 0);
735 count_from_all_detectors[d][v] = nodal_counts[d][dof_index];
736 }
737 }
738
739 search_position_in_reference_space(
740 count_from_all_detectors, experimental_counts, adjacent_cell,
741 tol_reference_location, max_cost, reference_location, position_found,
742 real_location);
743 }
744
745 if (position_found) {
746 found_positions.push_back(real_location);
747 return position_found;
748 } else {
749 position_found = find_position_global_search(experimental_counts,
750 tol_reference_location);
751 return position_found;
752 }
753}

755template <int dim> void RPTFEMReconstruction<dim>::trajectory() {
756 // Tolerance or the extrapolation limit in the reference space for a found
757 // position
758 double tol_reference_location =
759 fem_reconstruction_parameters.extrapolation_tolerance;
760
761 if (fem_reconstruction_parameters.mesh_type ==
762 Parameters::RPTFEMReconstructionParameters::FEMMeshType::dealii) {
763 if (fem_reconstruction_parameters.extrapolation_tolerance == 0.005) {
764 const unsigned int power = static_cast<unsigned int>(
765 pow(2, fem_reconstruction_parameters.mesh_refinement));
766 const unsigned int n_cell_z =
767 2 * fem_reconstruction_parameters.z_subdivisions * power;
768 tol_reference_location = parameters.reactor_height / n_cell_z * 1.15;
769 }
770 }
771
772 // Read and store all experimental counts
773 std::vector<std::vector<double>> all_experimental_counts;
774 {
775 TimerOutput::Scope t(computing_timer, "input_experimental_counts_file");
776 all_experimental_counts = read_detectors_counts<dim>(
777 fem_reconstruction_parameters.experimental_counts_file, n_detector);
778 }
779
780 {
781 TimerOutput::Scope t(computing_timer, "find_particle_positions");
782
783 if (fem_reconstruction_parameters.search_type ==
784 Parameters::RPTFEMReconstructionParameters::FEMSearchType::local) {
785 // It is set to "false" to force the global search on the first search
786 bool adjacent_cell_search = false;
787
788 // Find the position of the particle with the experimental counts by
789 // prioritizing local search
790 for (std::vector<double> &experimental_counts : all_experimental_counts) {
791 if (adjacent_cell_search)
792 adjacent_cell_search = find_position_local_search(
793 experimental_counts, tol_reference_location);
794 else
795 adjacent_cell_search = find_position_global_search(
796 experimental_counts, tol_reference_location);
797 }
798 } else {
799 // Find the position of the particle with the experimental counts by
800 // global search only
801 for (std::vector<double> &experimental_counts : all_experimental_counts) {
802 find_position_global_search(experimental_counts,
803 tol_reference_location);
804 }
805 }
806 }
807}
