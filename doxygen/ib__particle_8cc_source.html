7template <int dim> void IBParticle<dim>::initialize_all() {
8 // initialize all the variables associated to an immersed boundary particle.
9 radius = 1.;
10 particle_id = 0.;
11
12 inertia[0][0] = 1.;
13 inertia[0][1] = 0.;
14 inertia[0][2] = 0.;
15 inertia[1][0] = 0.;
16 inertia[1][1] = 1.;
17 inertia[1][2] = 0.;
18 inertia[2][0] = 0.;
19 inertia[2][1] = 0.;
20 inertia[2][2] = 1.;
21
22 fluid_forces[0] = 0.;
23 fluid_forces[1] = 0.;
24
25 velocity[0] = 0.;
26 velocity[1] = 0.;
27
28 position[0] = 0.;
29 position[1] = 0.;
30
31 fluid_torque[0] = 0.;
32 fluid_torque[1] = 0.;
33 fluid_torque[2] = 0.;
34
35 orientation[0] = 0.;
36 orientation[1] = 0.;
37 orientation[2] = 0.;
38
39 omega[0] = 0.;
40 omega[1] = 0.;
41 omega[2] = 0.;
42
43 if (dim == 3) {
44 fluid_forces[2] = 0.;
45 velocity[2] = 0.;
46 position[2] = 0.;
47 }
48
49 // Fill the vectors with default value
50 previous_fluid_forces = fluid_forces;
51 previous_fluid_torque = fluid_torque;
52 velocity_iter = velocity;
53
54 omega_iter = omega;
55 omega_impulsion = 0.;
56 omega_impulsion_iter = 0.;
57 impulsion = 0.;
58 impulsion_iter = 0.;
59 contact_impulsion = 0.;
60
61 previous_positions.resize(3);
62 previous_velocity.resize(3);
63 previous_orientation.resize(3);
64 previous_omega.resize(3);
65
66 for (unsigned int i = 0; i < 3; ++i) {
67 previous_positions[i] = position;
68 previous_velocity[i] = velocity;
69 previous_orientation[i] = orientation;
70 previous_omega[i] = omega;
71 }
72
73 f_position = std::make_shared<Functions::ParsedFunction<dim>>(dim);
74 f_velocity = std::make_shared<Functions::ParsedFunction<dim>>(dim);
75 f_omega = std::make_shared<Functions::ParsedFunction<dim>>(3);
76 f_orientation = std::make_shared<Functions::ParsedFunction<dim>>(3);
77
78 rotation_matrix[0][0] = 1.;
79 rotation_matrix[0][1] = 0.;
80 rotation_matrix[0][2] = 0.;
81 rotation_matrix[1][0] = 0.;
82 rotation_matrix[1][1] = 1.;
83 rotation_matrix[1][2] = 0.;
84 rotation_matrix[2][0] = 0.;
85 rotation_matrix[2][1] = 0.;
86 rotation_matrix[2][2] = 1.;
87}

122 properties[DEM::CFDDEMProperties::PropertiesIndex::dp] = radius * 2.0;

125 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_x] = omega[0];

127 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_z] = omega[2];
128 properties
129 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_x] =
130 fluid_forces[0];
131 properties
132 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_y] =
133 fluid_forces[1];
134 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_x] =
135 fluid_torque[0];
136 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_y] =
137 fluid_torque[1];
138 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_z] =
139 fluid_torque[2];
140 properties[DEM::CFDDEMProperties::PropertiesIndex::mass] = mass;

142 if (dim == 2) {
143 properties[DEM::CFDDEMProperties::PropertiesIndex::v_z] = 0.;
144 properties
145 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_z] =

147 }

171template <int dim>
172bool IBParticle<dim>::is_inside_crown(
173 const Point<dim> &evaluation_point, const double outer_radius,
174 const double inside_radius, const bool absolute_distance,
175 const typename DoFHandler<dim>::active_cell_iterator &cell_guess) {
176 const double effective_radius = shape->effective_radius;
177
178 double distance = shape->value_with_cell_guess(evaluation_point, cell_guess);
179 bool is_inside_outer_ring;
180 bool is_outside_inner_ring;
181 if (absolute_distance) {
182 is_inside_outer_ring = distance <= outer_radius;
183 is_outside_inner_ring = distance >= inside_radius;
184 } else {
185 is_inside_outer_ring = distance <= effective_radius * (outer_radius - 1);
186 is_outside_inner_ring = distance >= effective_radius * (inside_radius - 1);
187 }
188
189 return is_inside_outer_ring && is_outside_inner_ring;
190}

194 const double outer_radius,
195 const double inside_radius,

197 const double effective_radius = shape->effective_radius;
198

200 bool is_inside_outer_ring;
201 bool is_outside_inner_ring;
202 if (absolute_distance) {
203 is_inside_outer_ring = distance <= outer_radius;
204 is_outside_inner_ring = distance >= inside_radius;
205 } else {
206 is_inside_outer_ring = distance <= effective_radius * (outer_radius - 1);
207 is_outside_inner_ring = distance >= effective_radius * (inside_radius - 1);
208 }
209
210 return is_inside_outer_ring && is_outside_inner_ring;

240 DoFHandler<dim> &updated_dof_handler, const bool precalculations) {
241 if (integrate_motion || velocity.norm() > 1e-16 || omega.norm() > 1e-16) {
242 this->load_data_from_file();
