10namespace DEM {
11/* @brief Identify which type of solver is used by the DEM
12 * particles. This is used to identify which index of the ParticleHandler
13 * corresponds to which properties. Two types of solvers are currently
14 * supported. DEM implies pure DEM simulation whereas cfd_dem is used to
15 * indicate simulations in which the particles are coupled to CFD (CFD-DEM).
16 * In the latter case the particles carry additional properties related to the
17 * particle-fluid coupling.
18 */
19enum SolverType {
20 dem,
21 cfd_dem,
22 dem_mp,
23};
