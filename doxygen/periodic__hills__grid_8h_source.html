79/**
80 * @brief Constructor for the PeriodicHillsGrid.
81 *
82 * @param grid_arguments. A string with 5 parameters
83 * spacing_y : allows to control the shifting of horizontal line [0 - 1]
84 * alpha : allows to elongate while keeping the same flat region [0.5 - 3]
85 * repetitions_x : number of separation of cells in x before refinement
86 * repetitions_y : number of separation of cells in y before refinement
87 * repetitions_z : number of separation of cells in z before refinement
88 */
89
90template <int dim, int spacedim>
91PeriodicHillsGrid<dim, spacedim>::PeriodicHillsGrid(
92 const std::string &grid_arguments) {
93 this->grid_arguments = grid_arguments;
94
95 // Separate arguments of the string
96 std::vector<std::string> arguments;
97 std::stringstream s_stream(grid_arguments);
98 while (s_stream.good()) {
99 std::string substr;
100 getline(s_stream, substr, ';');
101 arguments.push_back(substr);
102 }
103
104 std::vector<double> arguments_double =
105 dealii::Utilities::string_to_double(arguments);
106 spacing_y = arguments_double[0];
107 alpha = arguments_double[1];
108 repetitions_x = static_cast<int>(arguments_double[2]);
109 repetitions_y = static_cast<int>(arguments_double[3]);
110 if (dim == 3)
111 repetitions_z = static_cast<int>(arguments_double[4]);
112
113 if (abs(alpha - 1) < 1e-6)
114 alpha = int(alpha);
115}

117/**
118 * @brief vector_value. This function is used to construct the geometry
119 * manifold. It changes the original point (op) of the transformed
120 * hyper_rectangle with hill_geometry to a new point (np) of the hill grid with
121 * per_hills_grid function.
122 *
123 * @param p. A point in space
124 *
125 * @param values. The vector of values which will be calculated at the position
126 * p.
127 */
128template <int dim, int spacedim>
129void PeriodicHillsPushForward<dim, spacedim>::vector_value(
130 const Point<spacedim> &op, Vector<double> &values) const {
131 const Point<spacedim> np =
132 PeriodicHillsGrid<dim, spacedim>::hill_geometry(op, spacing_y, alpha);
133
134 values(0) = np[0];
135 values(1) = np[1];
136
137 if (spacedim == 3)
138 values(2) = np[2];
139}

158/**
159 * \brief vector_value. This vector_value function is used to construct the
160 * geometry manifold. It changes the new point (np) of the hill grid to the
161 * original point (op) of the transformed hyper_rectangle grid. This function
162 * is mandatory to use FunctionManifold. First, it finds the minimum value of
163 * y depending the x position and then calculates the op with the inverse of
164 * the transformation done by hill_geometry function.
165 *
166 * \param np. A point in space.
167 *
168 * @param values. The vector of values which will be calculated at the position
169 * p.
170 */
171template <int dim, int spacedim>
172void PeriodicHillsPullBack<dim, spacedim>::vector_value(
173 const Point<spacedim> &np, Vector<double> &values) const {
174 double x = np[0];
175 double max_y = 3.035;
176 double max_x = 9.0;
177 double flat_region_length = 5.142;
178 double left_hill = 1.929;
179 double right_hill = 7.071;
180 double min_y;
181
182 // Reversing elongation and shifting of x lines
183 if (alpha != 1) {
184 if (x < left_hill * alpha)
185 x = (x / alpha);
186 else if (x > alpha * left_hill + flat_region_length)
187 x = (x - flat_region_length - alpha * left_hill) / alpha + right_hill;
188 else
189 x = x - (alpha * left_hill) + left_hill;
190 }
191
192 if (alpha > 1) {
193 if (x < max_x / 2)
194 x = (-(1 - 0.5) +
195 std::sqrt(std::pow((1 - 0.5), 2) - (4 * (1 / max_x) * -x))) /
196 (2 / max_x);
197 else if (x > max_x / 2 && x < max_x)
198 x = (-(1 + 1.5) + std::sqrt(std::pow((1 + 1.5), 2) -
199 (4 * (-1 / max_x) * (-0.5 * max_x - x)))) /
200 (2 * -1 / max_x);
201 }
202
203 // Reversing polynomial transformation and shifting of y lines
204 if (spacedim == 2) {
205 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
206 Point<spacedim>(x, 0), spacing_y, alpha)[1];
207 } else if (spacedim == 3) {
208 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
209 Point<spacedim>(x, 0, np[2]), spacing_y, alpha)[1];
210 values(2) = np[2];
211 }
212
213 double y = (np[1] - min_y) / (1 - min_y / max_y);
214
215 if (y < max_y / 2 && spacing_y != 0)
216 y = (-(1 - 0.5 * spacing_y) + std::sqrt(std::pow((1 - 0.5 * spacing_y), 2) -
217 (4 * (spacing_y / max_y) * -y))) /
218 (2 * spacing_y / max_y);
219 else if (y > max_y / 2 && y < max_y && spacing_y != 0)
220 y = (-(1 + 1.5 * spacing_y) + std::sqrt(std::pow((1 + 1.5 * spacing_y), 2) -
221 (4 * (-spacing_y / max_y) *
222 (-0.5 * spacing_y * max_y - y)))) /
223 (2 * -spacing_y / max_y);
224
225 values(0) = x;
226 values(1) = y;
227}

229/**
230 * @brief value. The value function does the same thing than vector_value for
231 * one component. This implementation is needed to use the gradient function
232 * inherited by AutoDerivativeFunction.
233 *
234 * @param p. An original point in space
235 *
236 * @param component. The component of the point (x=0, y=1, z=2)
237 *
238 */
239template <int dim, int spacedim>
240double PeriodicHillsPullBack<dim, spacedim>::value(
241 const Point<spacedim> &np, const unsigned int component) const {
242 double x = np[0];
243 double max_y = 3.035;
244 double max_x = 9.0;
245 double flat_region_length = 5.142;
246 double left_hill = 1.929;
247 double right_hill = 7.071;
248 double min_y;
249
250 if (alpha != 1) {
251 if (x < left_hill * alpha)
252 x = (x / alpha);
253 else if (x > alpha * left_hill + flat_region_length)
254 x = (x - flat_region_length - alpha * left_hill) / alpha + right_hill;
255 else
256 x = x - (alpha * left_hill) + left_hill;
257 }
258
259 if (alpha > 1) {
260 if (x < max_x / 2)
261 x = (-(1 - 0.5) +
262 std::sqrt(std::pow((1 - 0.5), 2) - (4 * (1 / max_x) * -x))) /
263 (2 / max_x);
264 else if (x > max_x / 2 && x <= max_x)
265 x = (-(1 + 1.5) + std::sqrt(std::pow((1 + 1.5), 2) -
266 (4 * (-1 / max_x) * (-0.5 * max_x - x)))) /
267 (2 * -1 / max_x);
268 }
269
270 if (spacedim == 2)
271 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
272 Point<spacedim>(x, 0), spacing_y, alpha)[1];
273 else if (spacedim == 3)
274 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
275 Point<spacedim>(x, 0, np[2]), spacing_y, alpha)[1];
276
277 double y = (np[1] - min_y) / (1 - min_y / max_y);
278
279 if (y < max_y / 2 && spacing_y != 0)
280 y = (-(1 - 0.5 * spacing_y) + std::sqrt(std::pow((1 - 0.5 * spacing_y), 2) -
281 (4 * (spacing_y / max_y) * -y))) /
282 (2 * spacing_y / max_y);
283 else if (y > max_y / 2 && y < max_y && spacing_y != 0)
284 y = (-(1 + 1.5 * spacing_y) + std::sqrt(std::pow((1 + 1.5 * spacing_y), 2) -
285 (4 * (-spacing_y / max_y) *
286 (-0.5 * spacing_y * max_y - y)))) /
287 (2 * -spacing_y / max_y);
288
289 Point<spacedim> op;
290 if (spacedim == 2)
291 op = {x, y};
292 else if (spacedim == 3)
293 op = {x, y, np[2]};
294
295 return op[component];
296}

298/**
299 * @brief The hill_geometry function calculates all the domain of the geometry
300 * with 6 polynomials depending the x position. (See Hill Geometry Definition
301 * file : https://turbmodels.larc.nasa.gov/Other_LES_Data/2dhill_periodic.html)
302 * This code has nondimensionalized geometry, but the coefficients provided
303 * need a hill height of 28.
304 * This function also does a gradual shifting of the horizontal lines
305 * prior to have smaller element on the bottom of the geometry where results
306 * are more important.
307 *
308 * @param p A point in space which will be adapted to the periodic hill geometry
309 *
310 * @param param Non-linear solver parameters
311 *
312 */
313template <int dim, int spacedim>
314Point<spacedim> PeriodicHillsGrid<dim, spacedim>::hill_geometry(
315 const Point<spacedim> &p, double spacing_y, double alpha) {
316 double H = 28; // Height dimension to use with polynomials
317 double x = p[0] * H, y = p[1] * H;
318 double max_y = 3.035 * H;
319 double max_x = 9.0 * H;
320 double flat_region_length = 5.142 * H;
321 double left_hill = 1.929 * H;
322 double right_hill = 7.071 * H;
323 double pos_x_left;
324 double pos_x_right;
325 double pos_y_bottom;
326 double pos_y_top;
327 double pos_y;
328 double y_0;
329 double diff;
330
331 // Gradual spacing and swifting depending on x position if the geometry
332 // is elongated
333 if (alpha > 1) {
334 if (x < max_x / 2) {
335 pos_x_left = x / -max_x + 0.5;
336 x -= pos_x_left * x;
337 } else if (x > max_x / 2 && x < max_x) {
338 pos_x_right = x / max_x - 0.5;
339 x += pos_x_right * (max_x - x);
340 }
341 }
342
343 // Polynomial coefficients :
344 const double a1 = 2.800000000000E+01, b1 = 0.000000000000E+00,
345 c1 = 6.775070969851E-03, d1 = -2.124527775800E-03;
346 const double a2 = 2.507355893131E+01, b2 = 9.754803562315E-01,
347 c2 = -1.016116352781E-01, d2 = 1.889794677828E-03;
348 const double a3 = 2.579601052357E+01, b3 = +8.206693007457E-01,
349 c3 = -9.055370274339E-02, d3 = 1.626510569859E-03;
350 const double a4 = 4.046435022819E+01, b4 = -1.379581654948E+00,
351 c4 = 1.945884504128E-02, d4 = -2.070318932190E-04;
352 const double a5 = 1.792461334664E+01, b5 = +8.743920332081E-01,
353 c5 = -5.567361123058E-02, d5 = 6.277731764683E-04;
354 const double a6 = 5.639011190988E+01, b6 = -2.010520359035E+00,
355 c6 = 1.644919857549E-02, d6 = 2.674976141766E-05;
356
357 double new_x = (max_x - x); // x for the left side of the geometry
358
359 // Gradual spacing and swifting depending on y position
360 if (y < max_y / 2 && (std::abs(spacing_y) > 0)) {
361 pos_y_bottom = y / -max_y + 0.5;
362 y -= spacing_y * pos_y_bottom * y;
363 } else if (y > max_y / 2 && y < max_y && (std::abs(spacing_y) > 0)) {
364 pos_y_top = y / max_y - 0.5;
365 y += spacing_y * pos_y_top * (max_y - y);
366 }
367
368 pos_y = y / -max_y + 1;
369
370 // Polynomial equations :
371 if (x >= 0 && x < 9) {
372 y += pos_y * (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
373
374 // Checking if y is under 28 and correction
375 y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
376 diff = y_0 - 28.0;
377 if (diff > 0)
378 y -= pos_y * diff;
379 } else if (x >= 9 && x < 14)
380 y += pos_y * (a2 + b2 * x + c2 * std::pow(x, 2) + d2 * std::pow(x, 3));
381 else if (x >= 14 && x < 20)
382 y += pos_y * (a3 + b3 * x + c3 * std::pow(x, 2) + d3 * std::pow(x, 3));
383 else if (x >= 20 && x < 30)
384 y += pos_y * (a4 + b4 * x + c4 * std::pow(x, 2) + d4 * std::pow(x, 3));
385 else if (x >= 30 && x < 40)
386 y += pos_y * (a5 + b5 * x + c5 * std::pow(x, 2) + d5 * std::pow(x, 3));
387 else if (x >= 40 && x < 54) {
388 y += pos_y * (a6 + b6 * x + c6 * std::pow(x, 2) + d6 * std::pow(x, 3));
389 if (y < 0)
390 y = 0;
391 } else if (x <= 252 && x > 243) {
392 y += pos_y *
393 (a1 + b1 * new_x + c1 * std::pow(new_x, 2) + d1 * std::pow(new_x, 3));
394 y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
395 diff = y_0 - 28.0;
396 if (diff > 0)
397 y -= pos_y * diff;
398 } else if (x <= 243 && x > 238)
399 y += pos_y *
400 (a2 + b2 * new_x + c2 * std::pow(new_x, 2) + d2 * std::pow(new_x, 3));
401 else if (x <= 238 && x > 232)
402 y += pos_y *
403 (a3 + b3 * new_x + c3 * std::pow(new_x, 2) + d3 * std::pow(new_x, 3));
404 else if (x <= 232 && x > 222)
405 y += pos_y *
406 (a4 + b4 * new_x + c4 * std::pow(new_x, 2) + d4 * std::pow(new_x, 3));
407 else if (x <= 222 && x > 212)
408 y += pos_y *
409 (a5 + b5 * new_x + c5 * std::pow(new_x, 2) + d5 * std::pow(new_x, 3));
410 else if (x <= 212 && x > 198) {
411 y += pos_y *
412 (a6 + b6 * new_x + c6 * std::pow(new_x, 2) + d6 * std::pow(new_x, 3));
413 if (y < 0)
414 y = 0;
415 } else
416 y += 0;
417
418 // Elongation of the geometry with the alpha factor
419 // Note : The length of the flat region is always the same length
420 if (x < left_hill)
421 x = alpha * x;
422 else if (x > right_hill)
423 x = alpha * (x - right_hill) + flat_region_length + alpha * left_hill;
424 else
425 x = (x - left_hill) + (alpha * left_hill);
426
427 Point<spacedim> q;
428 q[0] = (x / H);
429 q[1] = (y / H);
430
431 if (spacedim == 3)
432 q[2] = p[2];
433
434 return q;
435}

437/**
438 * @brief make_grid. The make_grid function generates a hyper rectangle of the
439 * size of the domain and then transforms it to the hill geometry. It also
440 * constructs the geometry manifold with FunctionManifold and finally sets the
441 * manifold.
442 *
443 * @param triangulation. The triangulation object on which the grid is generated
444 */
445template <int dim, int spacedim>
446void PeriodicHillsGrid<dim, spacedim>::make_grid(
447 Triangulation<dim, spacedim> &triangulation) {
448 std::vector<unsigned int> repetitions(2);
449 repetitions[0] = repetitions_x;
450 repetitions[1] = repetitions_y;
451
452 if (dim == 2) {
453 if (alpha != 1 && (repetitions_x > 1 || repetitions_y > 1))
454 throw std::logic_error("When alpha parameter is not 1, repetition "
455 "parameters should be all set to 1");
456
457 GridGenerator::subdivided_hyper_rectangle(triangulation, repetitions,
458 Point<dim>(0.0, 0.0),
459 Point<dim>(9.0, 3.035), true);
460 } else if (dim == 3) {
461 if (alpha != 1 &&
462 (repetitions_x > 1 || repetitions_y > 1 || repetitions_z > 1))
463 throw std::logic_error("When alpha parameter is not 1, repetition "
464 "parameters should be all set to 1");
465 repetitions.push_back(repetitions_z);
466 GridGenerator::subdivided_hyper_rectangle(
467 triangulation, repetitions, Point<dim>(0.0, 0.0, 0.0),
468 Point<dim>(9.0, 3.035, 4.5), true);
469 }
470
471 // Transformation of the geometry with the hill geometry
472 GridTools::transform(
473 [this](const Point<spacedim> &p) {
474 return this->hill_geometry(p, spacing_y, alpha);
475 },
476 triangulation);
477
478 // Manifold construction
479 static const FunctionManifold<dim, spacedim, spacedim> manifold_func(
480 std::make_unique<PeriodicHillsPushForward<dim, spacedim>>(spacing_y,
481 alpha),
482 std::make_unique<PeriodicHillsPullBack<dim, spacedim>>(spacing_y, alpha));
483 triangulation.set_manifold(1, manifold_func);
484 triangulation.set_all_manifold_ids(1);
485}
