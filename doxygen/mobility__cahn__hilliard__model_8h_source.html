129/**
130 * @brief Quartic mobility model.
131 *
132 * The mobility function is the following : \f$M(\phi) = D(1-\phi^2)^2 \f$
133 * where D is the mobility constant.
134 */
135class MobilityCahnHilliardModelQuartic : public MobilityCahnHilliardModel {
136public:
137 /**
138 * @brief Default constructor.
139 *
140 * @param[in] p_mobility_cahn_hilliard_constant the user defined mobility
141 * constant.
142 */
143 MobilityCahnHilliardModelQuartic(
144 const double p_mobility_cahn_hilliard_constant)
145 : mobility_cahn_hilliard_constant(p_mobility_cahn_hilliard_constant) {
146 this->model_depends_on[field::phase_order_cahn_hilliard] = true;
147 }
148
149 /**
150 * @brief Destructor of derived class.
151 */
152 ~MobilityCahnHilliardModelQuartic() override = default;
153
154 /**
155 * @brief Return the mobility constant.
156 * @return Value of the mobility constant.
157 */
158 double get_mobility_constant() override {
159 return mobility_cahn_hilliard_constant;
160 }
161
162 /**
163 * @brief Calculate the mobility.
164 * @param[in] fields_value Value of the various fields on which the mobility
165 * may depend.
166 * @return Value of the mobility calculated with the fields_value.
167 */
168 double value(const std::map<field, double> &fields_value) override {
169 Assert(
170 fields_value.contains(field::phase_order_cahn_hilliard),
171 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
172 "phase_order_cahn_hilliard"));
173 const double &phase_order_cahn_hilliard =
174 fields_value.at(field::phase_order_cahn_hilliard);
175
176 // The phase order values are clamped to avoid unphysical mobilities in the
177 // bulk phases.
178 return mobility_cahn_hilliard_constant *
179 std::pow((1 - std::min(1.0, phase_order_cahn_hilliard *
180 phase_order_cahn_hilliard)),
181 2);
182 }
183
184 /**
185 * @brief Calculate the vector of mobility.
186 * @param[in] field_vectors Vectors of the fields on which the mobility
187 * may depend.
188 * @param[out] property_vector Vector of the mobility values.
189 */
190 void vector_value(const std::map<field, std::vector<double>> &field_vectors,
191 std::vector<double> &property_vector) override {
192 Assert(
193 field_vectors.contains(field::phase_order_cahn_hilliard),
194 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
195 "phase_order_cahn_hilliard"));
196 const std::vector<double> &phase_order_cahn_hilliard =
197 field_vectors.at(field::phase_order_cahn_hilliard);
198 for (unsigned int i = 0; i < property_vector.size(); ++i) {
199 property_vector[i] =
200 mobility_cahn_hilliard_constant *
201 std::pow((1 - std::min(1.0, phase_order_cahn_hilliard[i] *
202 phase_order_cahn_hilliard[i])),
203 2);
204 }
205 }
206
207 /**
208 * @brief Calculate the jacobian (the partial derivative) of the
209 * mobility with respect to a field.
210 * @param[in] fields_value Value of the various fields on which the mobility
211 * may depend.
212 * @param[in] id Indicator of the field with respect to which the Jacobian
213 * should be calculated.
214 * @return Value of the partial derivative of the mobility with respect to
215 * the field.
216 */
217
218 double jacobian(const std::map<field, double> &fields_value,
219 [[maybe_unused]] field id) override {
220 Assert(
221 fields_value.contains(field::phase_order_cahn_hilliard),
222 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
223 "phase_order_cahn_hilliard"));
224 const double &phase_order_cahn_hilliard =
225 fields_value.at(field::phase_order_cahn_hilliard);
226
227 return -4 *
228 (std::max(-1.0, std::min(phase_order_cahn_hilliard, 0.0)) +
229 std::min(1.0, std::max(phase_order_cahn_hilliard, 0.0))) *
230 mobility_cahn_hilliard_constant *
231 (1 - std::min(1.0, phase_order_cahn_hilliard *
232 phase_order_cahn_hilliard));
233 }
234
235 /**
236 * @brief Calculate the derivative of the mobility with respect to a field.
237 * @param[in] fields_vectors Vector for the values of the fields used to
238 * evaluate the mobility.
239 * @param[in] id Identifier of the field with respect to which a derivative
240 * should be calculated.
241 * @param[out] jacobian_vector Vector of the values of the derivatives of the
242 * mobility with respect to the field id.
243 */
244
245 void
246 vector_jacobian(const std::map<field, std::vector<double>> &fields_vectors,
247 [[maybe_unused]] const field id,
248 std::vector<double> &jacobian_vector) override {
249 Assert(
250 fields_vectors.contains(field::phase_order_cahn_hilliard),
251 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
252 "phase_order_cahn_hilliard"));
253 const std::vector<double> &phase_order_cahn_hilliard =
254 fields_vectors.at(field::phase_order_cahn_hilliard);
255 for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
256 jacobian_vector[i] =
257 -4 *
258 (std::max(-1.0, std::min(phase_order_cahn_hilliard[i], 0.0)) +
259 std::min(1.0, std::max(phase_order_cahn_hilliard[i], 0.0))) *
260 mobility_cahn_hilliard_constant *
261 (1 - std::min(1.0, phase_order_cahn_hilliard[i] *
262 phase_order_cahn_hilliard[i]));
263 }
264
265private:
266 const double mobility_cahn_hilliard_constant;
267};
