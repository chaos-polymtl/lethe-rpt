81void LagrangianPhysicalProperties::parse_parameters(ParameterHandler &prm) {
82 prm.enter_subsection("lagrangian physical properties");
83 initialize_containers(
84 particle_average_diameter, particle_size_std, distribution_type,
85 particle_custom_diameter, particle_custom_probability,
86 seed_for_distributions, number, density_particle, youngs_modulus_particle,
87 poisson_ratio_particle, restitution_coefficient_particle,
88 friction_coefficient_particle,
89 rolling_viscous_damping_coefficient_particle,
90 rolling_friction_coefficient_particle, surface_energy_particle,
91 hamaker_constant_particle, thermal_conductivity_particle,
92 specific_heat_particle, microhardness_particle, surface_slope_particle,
93 surface_roughness_particle, thermal_accommodation_particle,
94 real_youngs_modulus_particle);
95
96 // Deprecated parameter handling
97 // <g> used to be 3 parameters: <gx>, <gy> and <gz>
98 // If <gx> is in the input file, it will be used as the value for <g>
99 // as an alias. This way, the parameter <g> is not a tensor and allows the
100 // parsing of deprecated parameters.
101 g = value_string_to_tensor<3>(prm.get("g"), prm.get_double("gy"),
102 prm.get_double("gz"));
103
104 particle_type_number = prm.get_integer("number of particle types");
105
106 for (unsigned int id = 0; id < particle_type_number; ++id) {
107 prm.enter_subsection("particle type " + Utilities::int_to_string(id, 1));
108 parse_particle_properties(id, prm);
109 prm.leave_subsection();
110 }
111
112 youngs_modulus_wall = prm.get_double("young modulus wall");
113 poisson_ratio_wall = prm.get_double("poisson ratio wall");
114 restitution_coefficient_wall = prm.get_double("restitution coefficient wall");
115 friction_coefficient_wall = prm.get_double("friction coefficient wall");
116 rolling_friction_wall = prm.get_double("rolling friction wall");
117 rolling_viscous_damping_wall = prm.get_double("rolling viscous damping wall");
118 surface_energy_wall = prm.get_double("surface energy wall");
119 hamaker_constant_wall = prm.get_double("hamaker constant wall");
120 thermal_conductivity_wall = prm.get_double("thermal conductivity wall");
121 microhardness_wall = prm.get_double("microhardness wall");
122 surface_slope_wall = prm.get_double("surface slope wall");
123 surface_roughness_wall = prm.get_double("surface roughness wall");
124 thermal_accommodation_wall = prm.get_double("thermal accommodation wall");
125 real_youngs_modulus_wall = prm.get_double("real young modulus wall");
126 // Only use the real Young's modulus if it is higher than the Young's
127 // modulus
128 if (real_youngs_modulus_wall < youngs_modulus_wall)
129 real_youngs_modulus_wall = youngs_modulus_wall;
130
131 thermal_conductivity_gas = prm.get_double("thermal conductivity gas");
132 specific_heat_gas = prm.get_double("specific heat gas");
133 dynamic_viscosity_gas = prm.get_double("dynamic viscosity gas");
134 specific_heats_ratio_gas = prm.get_double("specific heats ratio gas");
135 molecular_mean_free_path_gas = prm.get_double("molecular mean free path gas");
136
137 prm.leave_subsection();
138}

195void LagrangianPhysicalProperties::parse_particle_properties(
196 const unsigned int &particle_type, const ParameterHandler &prm) {
197 particle_average_diameter.at(particle_type) = prm.get_double("diameter");
198 particle_size_std.at(particle_type) = prm.get_double("standard deviation");
199 particle_custom_diameter.at(particle_type) =
200 convert_string_to_vector<double>(prm, "custom diameters");
201 particle_custom_probability.at(particle_type) =
202 convert_string_to_vector<double>(prm, "custom volume fractions");
203 seed_for_distributions.push_back(prm.get_integer("random seed distribution"));
204
205 double probability_sum =
206 std::reduce(particle_custom_probability.at(particle_type).begin(),
207 particle_custom_probability.at(particle_type).end());
208
209 // We make sure that the cumulative probability is equal to 1.
210 if (std::abs(probability_sum - 1.0) > 1.e-5) {
211 throw(std::runtime_error("Invalid custom volume fraction. The sum of "
212 "volume fractions should be equal to 1.0 "));
213 }
214 const std::string size_distribution_type_str =
215 prm.get("size distribution type");
216 if (size_distribution_type_str == "uniform") {
217 distribution_type.at(particle_type) = SizeDistributionType::uniform;
218 } else if (size_distribution_type_str == "normal") {
219 distribution_type.at(particle_type) = SizeDistributionType::normal;
220 } else if (size_distribution_type_str == "custom") {
221 distribution_type.at(particle_type) = SizeDistributionType::custom;
222 } else {
223 throw(std::runtime_error("Invalid size distribution type. Choices are "
224 "<uniform|normal|custom>."));
225 }
226 number.at(particle_type) = prm.get_integer("number of particles");
227 density_particle.at(particle_type) = prm.get_double("density particles");
228 youngs_modulus_particle.at(particle_type) =
229 prm.get_double("young modulus particles");
230 poisson_ratio_particle.at(particle_type) =
231 prm.get_double("poisson ratio particles");
232 restitution_coefficient_particle.at(particle_type) =
233 prm.get_double("restitution coefficient particles");
234 friction_coefficient_particle.at(particle_type) =
235 prm.get_double("friction coefficient particles");
236 rolling_viscous_damping_coefficient_particle.at(particle_type) =
237 prm.get_double("rolling viscous damping particles");
238 rolling_friction_coefficient_particle.at(particle_type) =
239 prm.get_double("rolling friction particles");
240 surface_energy_particle.at(particle_type) =
241 prm.get_double("surface energy particles");
242 hamaker_constant_particle.at(particle_type) =
243 prm.get_double("hamaker constant particles");
244 thermal_conductivity_particle.at(particle_type) =
245 prm.get_double("thermal conductivity particles");
246 specific_heat_particle.at(particle_type) =
247 prm.get_double("specific heat particles");
248 microhardness_particle.at(particle_type) =
249 prm.get_double("microhardness particles");
250 surface_slope_particle.at(particle_type) =
251 prm.get_double("surface slope particles");
252 surface_roughness_particle.at(particle_type) =
253 prm.get_double("surface roughness particles");
254 thermal_accommodation_particle.at(particle_type) =
255 prm.get_double("thermal accommodation particles");
256 real_youngs_modulus_particle.at(particle_type) =
257 prm.get_double("real young modulus particles");
258 // Only use the real Young's modulus if it is higher than the Young's
259 // modulus
260 if (real_youngs_modulus_particle.at(particle_type) <
261 youngs_modulus_particle.at(particle_type)) {
262 real_youngs_modulus_particle.at(particle_type) =
263 youngs_modulus_particle.at(particle_type);
264 }
265}

425template <int dim>
426void InsertionInfo<dim>::parse_parameters(ParameterHandler &prm) {
427 prm.enter_subsection("insertion info");
428 {
429 const std::string insertion = prm.get("insertion method");
430 if (insertion == "file")
431 insertion_method = InsertionMethod::file;
432 else if (insertion == "plane")
433 insertion_method = InsertionMethod::plane;
434 else if (insertion == "list")
435 insertion_method = InsertionMethod::list;
436 else if (insertion == "volume")
437 insertion_method = InsertionMethod::volume;
438 else {
439 throw(std::runtime_error("Invalid insertion method "));
440 }
441 inserted_this_step =
442 prm.get_integer("inserted number of particles at each time step");
443 insertion_frequency = prm.get_integer("insertion frequency");
444
445 // Clear:
446 removing_particles_in_region = prm.get_bool("remove particles");
447
448 const std::vector<std::string> removal_box_point_coordinates_list(
449 Utilities::split_string_list(prm.get("removal box points coordinates"),
450 ":"));
451
452 std::vector<double> removal_point_coord_temp_1 =
453 Utilities::string_to_double(Utilities::split_string_list(
454 removal_box_point_coordinates_list.at(0)));
455 std::vector<double> removal_point_coord_temp_2 =
456 Utilities::string_to_double(Utilities::split_string_list(
457 removal_box_point_coordinates_list.at(1)));
458
459 if (removal_point_coord_temp_1.size() == 2 &&
460 removal_point_coord_temp_2.size() == 2) {
461 removal_point_coord_temp_1.push_back(0.);
462 removal_point_coord_temp_2.push_back(0.);
463 }
464
465 for (unsigned int i = 0; i < 3; ++i) {
466 clear_box_point_1[i] = removal_point_coord_temp_1.at(i);
467 clear_box_point_2[i] = removal_point_coord_temp_2.at(i);
468 }
469
470 // File:
471 // File for the insertion
472 list_of_input_files =
473 convert_string_to_vector<std::string>(prm, "list of input files");
474
475 // Plane:
476 // Insertion plane normal vector
477 insertion_plane_normal_vector =
478 value_string_to_tensor<3>(prm.get("insertion plane normal vector"));
479
480 // Insertion plane point
481 insertion_plane_point =
482 value_string_to_tensor<3>(prm.get("insertion plane point"));
483
484 // List:
485 // Read x, y and z lists
486 list_x = convert_string_to_vector<double>(prm, "list x");
487 list_y = convert_string_to_vector<double>(prm, "list y");
488 list_z = convert_string_to_vector<double>(prm, "list z");
489
490 // Find which vector is the longest
491 int max_size = std::max({list_x.size(), list_y.size(), list_z.size()});
492
493 // Resize the vectors in the case that one was longer
494 list_x.resize(max_size);
495 list_y.resize(max_size);
496 list_z.resize(max_size);
497
498 // Read vx, vv and vz
499 list_vx = convert_string_to_vector<double>(prm, "list velocity x");
500 list_vy = convert_string_to_vector<double>(prm, "list velocity y");
501 list_vz = convert_string_to_vector<double>(prm, "list velocity z");
502
503 // Fill the velocity vectors with zeros to match the size of list_x
504 if (list_vx != list_x)
505 list_vx.resize(max_size);
506 if (list_vy != list_x)
507 list_vy.resize(max_size);
508 if (list_vz != list_x)
509 list_vz.resize(max_size);
510
511 // Read wx, wy and wz
512 list_wx = convert_string_to_vector<double>(prm, "list omega x");
513 list_wy = convert_string_to_vector<double>(prm, "list omega y");
514 list_wz = convert_string_to_vector<double>(prm, "list omega z");
515
516 // Fill the angular velocity vectors with zeros to match the size of
517 // list_x
518 if (list_wx != list_x)
519 list_wx.resize(max_size);
520 if (list_wy != list_x)
521 list_wy.resize(max_size);
522 if (list_wz != list_x)
523 list_wz.resize(max_size);
524
525 // Read the diameters list
526 list_d = convert_string_to_vector<double>(prm, "list diameters");
527
528 // Read the temperatures list
529 list_T = convert_string_to_vector<double>(prm, "list temperatures");
530
531 // Volume:
532 std::vector<int> axis_order =

1453void ParticleRayTracing<dim>::parse_parameters(ParameterHandler &prm) {
1454 prm.enter_subsection("particle ray tracing");
1455 {
1456 // Location of the first photon to be inserted
1457 starting_point = point_nd_to_3d(Point<dim>(value_string_to_tensor<dim>(

1462 Utilities::split_string_list(prm.get("insertion unit tensors"), ":"));

1464 Utilities::split_string_list(
